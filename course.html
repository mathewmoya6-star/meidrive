END, '1-2 hours' ) ON CONFLICT (course_id, unit_number) DO NOTHING; END LOOP; END LOOP; END $$; -- Verify data SELECT 'courses' as table_name, COUNT(*) as count FROM public.courses UNION ALL SELECT 'course_units', COUNT(*) FROM public.course_units UNION ALL SELECT 'user_progress', COUNT(*) FROM public.user_progress;
